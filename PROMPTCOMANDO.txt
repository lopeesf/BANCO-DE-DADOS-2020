MODELO FISÍCO

CREATE TABLE CLIENTES (

CPF CHAR(11),
NOME VARCHAR(50),
CELULAR CHAR(12),
DATA_NASC DATE,
RENDA DECIMAL(6,2),
RG CHAR(15),
CNH CHAR(15),

PRIMARY KEY (CPF));







CREATE TABLE VEICULOS(

PLACA CHAR(10),
MARCA VARCHAR(20),
MODELO VARCHAR(20),
COMBUSTIVEL CHAR(3),
COR VARCHAR(10),
ANO_MOD CHAR(4),
ANO_FAB CHAR(4),
DIARIA DECIMAL(6,2),
SEGURO CHAR(1),
PRIMARY KEY (PLACA));


CREATE TABLE LOCACAO (
FK_CLIENTES_CPF CHAR(11),
FK_VEICULOS_PLACA CHAR(10),
DATA_RET DATE,
DATA_ENT DATE,

FOREIGN KEY (FK_CLIENTES_CPF)
  REFERENCES CLIENTES (CPF),

FOREIGN KEY (FK_VEICULOS_PLACA)
  REFERENCES VEICULOS (PLACA));
